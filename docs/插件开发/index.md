│  admin.php --------------------------------------------------------------------后台接口入口
│  api.php --------------------------------------------------------------------手机端接口入口
│  console.php --------------------------------------------------------------------控制台入口
│  frontend.php --------------------------------------------------------------------页面端入口

│  upgrade.php --------------------------------------------------------------------更新入口

│  │  DefaultController.php

├─config --------------------------------------------------------------------配置目录
│      account.php --------------------------------------------------------------------自定义配置
│      admin.php --------------------------------------------------------------------后台接口配置
│      api.php --------------------------------------------------------------------手机端接口配置
│      frontend.php --------------------------------------------------------------------页面端配置
│      menu.php --------------------------------------------------------------------菜单路由配置

│      DocController.php
├─migrations --------------------------------------------------------------------数据库脚本目录，分版本
